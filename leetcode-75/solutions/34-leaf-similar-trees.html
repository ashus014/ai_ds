872. Leaf-Similar Trees

LeetCode 75 #34 — Record leaf values left-to-right in each tree, then compare lists. Preorder is natural; inorder and postorder (left before right) visit leaves in the same order.

LeetCode 75 / Binary Tree — DFS

Hint

pre-order DFS; collect both nodes and compare

Code

Three full solutions; each has its own Copy all. Preorder first (✅); then inorder and postorder variants.

✅ Preorder

Solution.java — preorder 
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
import java.util.*;

class Solution {
    public boolean leafSimilar(TreeNode root1, TreeNode root2) {
        List<Integer> leaves1 = new ArrayList<>();
        List<Integer> leaves2 = new ArrayList<>();

        preOrder(root1, leaves1);
        preOrder(root2, leaves2);

        return leaves1.equals(leaves2);
    }

    private void preOrder(TreeNode root, List<Integer> leaves) {

        if (root == null) {
            return;
        }

        if (root.left == null && root.right == null) {
            leaves.add(root.val);
        }

        preOrder(root.left, leaves);
        preOrder(root.right, leaves);
    }
}

Inorder

Solution.java — inorder 
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
import java.util.*;

class Solution {
    public boolean leafSimilar(TreeNode root1, TreeNode root2) {
        List<Integer> leaves1 = new ArrayList<>();
        List<Integer> leaves2 = new ArrayList<>();

        inOrder(root1, leaves1);
        inOrder(root2, leaves2);

        return leaves1.equals(leaves2);
    }

    private void inOrder(TreeNode root, List<Integer> leaves) {
        if (root == null) {
            return;
        }

        inOrder(root.left, leaves);

        if (root.left == null && root.right == null) {
            leaves.add(root.val);
        }

        inOrder(root.right, leaves);
    }
}

Postorder

Solution.java — postorder 
/**
 * Definition for a binary tree node.
 * public class TreeNode {
 *     int val;
 *     TreeNode left;
 *     TreeNode right;
 *     TreeNode() {}
 *     TreeNode(int val) { this.val = val; }
 *     TreeNode(int val, TreeNode left, TreeNode right) {
 *         this.val = val;
 *         this.left = left;
 *         this.right = right;
 *     }
 * }
 */
import java.util.*;

class Solution {
    public boolean leafSimilar(TreeNode root1, TreeNode root2) {
        List<Integer> leaves1 = new ArrayList<>();
        List<Integer> leaves2 = new ArrayList<>();

        postOrder(root1, leaves1);
        postOrder(root2, leaves2);

        return leaves1.equals(leaves2);
    }

    private void postOrder(TreeNode root, List<Integer> leaves) {
        if (root == null) {
            return;
        }

        postOrder(root.left, leaves);
        postOrder(root.right, leaves);

        if (root.left == null && root.right == null) {
            leaves.add(root.val);
        }
    }
}
