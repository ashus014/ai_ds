399. Evaluate Division

LeetCode 75 #46 — Model a / b = v as weighted edges: a → b with weight v, and b → a with 1/v. A query x / y is the product of edge weights along any path from x to y; use visited to avoid cycles. If x is missing from the graph, return -1.0.

LeetCode 75 / Graphs — DFS

Hint

Map<String, Map<String, Double>> for neighbors and ratios. For each equation, add both directions. DFS from start with running product; when start.equals(end), store product in a one-element double[] (mutable “out” parameter).

class Solution {
    public double[] calcEquation(List<List<String>> equations, double[] values, List<List<String>> queries) {
        Map<String, Map<String, Double>> graph = new HashMap<>();
        double[] results = new double[queries.size()];

        for (int i = 0; i < equations.size(); i++) {
            String var1 = equations.get(i).get(0);
            String var2 = equations.get(i).get(1);

            double value = values[i];

            graph.putIfAbsent(var1, new HashMap<>());
            graph.putIfAbsent(var2, new HashMap<>());
            graph.get(var1).put(var2, value);
            graph.get(var2).put(var1, 1 / value);
        }

        for (int i = 0; i < queries.size(); i++) {
            String start = queries.get(i).get(0);
            String end = queries.get(i).get(1);
            double[] answer = new double[]{-1.0};
            Double product = 1.0;

            if (graph.containsKey(start)) {
                Set<String> visited = new HashSet<>();
                dfs(graph, start, end, visited, product, answer);
            }

            results[i] = answer[0];
        }
        return results;
    }

    private void dfs(Map<String, Map<String, Double>> graph, String start, String end, Set<String> visited, Double product, double[] answer) {
        if (visited.contains(start)) {
            return;
        }

        visited.add(start);

        if (start.equals(end)) {
            answer[0] = product;
            return;
        }

        for (Map.Entry<String, Double> neighbour : graph.get(start).entrySet()) {
            String v = neighbour.getKey();
            double val = neighbour.getValue();
            dfs(graph, v, end, visited, product * val, answer);
        }
    }
}
