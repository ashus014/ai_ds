Topological sort

DFS stack order and Kahn's BFS (DAG)

📅 April 3, 2026

DFS approach

April 3, 2026

Build an adjacency list, run DFS, push nodes to a stack after visiting descendants, then pop to get topological order.

AshuSinghIN 
import java.util.*;

class Solution {

    public ArrayList<Integer> topoSort(int V, int[][] edges) {

        // Step 1: Build adjacency list
        ArrayList<ArrayList<Integer>> adj = new ArrayList<>();
        for (int i = 0; i < V; i++) {
            adj.add(new ArrayList<>());
        }

        for (int[] edge : edges) {
            int u = edge[0];
            int v = edge[1];
            adj.get(u).add(v);
        }

        // Step 2: visited array + stack
        boolean[] visited = new boolean[V];
        Stack<Integer> st = new Stack<>();

        // Step 3: DFS on all nodes
        for (int i = 0; i < V; i++) {
            if (!visited[i]) {
                dfs(i, adj, visited, st);
            }
        }

        // Step 4: pop stack to get topo order
        ArrayList<Integer> result = new ArrayList<>();
        while (!st.isEmpty()) {
            result.add(st.pop());
        }

        return result;
    }

    void dfs(int u, ArrayList<ArrayList<Integer>> adj, boolean[] visited, Stack<Integer> st) {

        visited[u] = true;

        // Children pushed first
        for (int v : adj.get(u)) {
            if (!visited[v]) {
                dfs(v, adj, visited, st);
            }
        }

        // then the parent
        st.push(u);
    }
}

Kahn's algorithm

April 3, 2026

Topological sort done via BFS is Kahn's algorithm.

AshuSinghIN 
import java.util.*;

class Solution {

    public ArrayList<Integer> topoSort(int V, int[][] edges) {

        // Step 1: adjacency list
        ArrayList<ArrayList<Integer>> adj = new ArrayList<>();
        for (int i = 0; i < V; i++) {
            adj.add(new ArrayList<>());
        }

        // Step 2: indegree array
        int[] indegree = new int[V];

        for (int[] edge : edges) {
            int u = edge[0];
            int v = edge[1];

            adj.get(u).add(v);
            indegree[v]++;  // important
        }

        // Step 3: queue for BFS
        Queue<Integer> q = new LinkedList<>();

        for (int i = 0; i < V; i++) {
            if (indegree[i] == 0) {
                q.offer(i);
            }
        }

        // Step 4: BFS
        ArrayList<Integer> result = new ArrayList<>();

        while (!q.isEmpty()) {
            int u = q.poll();
            result.add(u);

            for (int v : adj.get(u)) {
                indegree[v]--;

                if (indegree[v] == 0) {
                    q.offer(v);
                }
            }
        }

        return result;
    }
}
