Prim's Algorithm for Minimum Spanning Tree (MST)

Greedy construction of an MST: grow a single tree by always adding the least expensive edge to a node not yet in the tree.

📅 April 4, 2026

Prim's algorithm

  • Start from any node and treat it as part of the tree.
  • At each step, pick the cheapest edge connecting the tree to a new node.
  • Keep expanding until all nodes are connected (no cycles, minimum total cost).

Code

Two styles: int[] in the priority queue, or a small P class. Copy either block from the toolbar.

PriorityQueue with int[] {weight, node}

import java.util.*;

class Solution {

    // Function to find sum of weights of edges of the Minimum Spanning Tree.
    int spanningTree(int V, int[][] edges) {

        // Step 1: Create adjacency list
        // adj.get(u) will contain pairs { neighbor, weight }
        ArrayList<ArrayList<int[]>> adj = new ArrayList<>();

        // Initialize empty list for each node
        for (int i = 0; i < V; i++) {
            adj.add(new ArrayList<>());
        }

        // Fill adjacency list from edges input
        for (int[] e : edges) {
            int u = e[0];       // source node
            int v = e[1];       // destination node
            int wt = e[2];      // edge weight

            // Since graph is undirected, add both directions
            adj.get(u).add(new int[]{v, wt});
            adj.get(v).add(new int[]{u, wt});
        }

        // Step 2: Min heap — entries { weight, node }; smallest weight first
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[0] - b[0]);

        // Start from node 0 with weight 0
        pq.add(new int[]{0, 0});

        // Track which nodes are already included in MST
        boolean[] inMST = new boolean[V];

        int sum = 0; // total weight of MST

        // Step 3: Prim's Algorithm
        while (!pq.isEmpty()) {

            int[] p = pq.poll();

            int wt = p[0];      // edge weight to reach this node
            int node = p[1];    // current node

            // If node already included, skip (avoid cycles)
            if (inMST[node] == true)
                continue;

            // Include this node in MST
            inMST[node] = true;

            // Add its edge weight to total MST cost
            sum += wt;

            // Explore all neighbors of current node
            for (int[] tmp : adj.get(node)) {

                int neighbor = tmp[0];      // adjacent node
                int neighbor_wt = tmp[1];   // weight of edge

                // If neighbor not yet in MST, add to PQ
                if (inMST[neighbor] == false) {
                    pq.add(new int[]{neighbor_wt, neighbor});
                }
            }
        }

        // Return total weight of MST
        return sum;
    }
}

Pair class P in the priority queue

import java.util.*;

class Solution {

    // Pair class -> stores {weight, node}
    static class P {
        int weight;
        int node;

        P(int weight, int node) {
            this.weight = weight;
            this.node = node;
        }
    }

    public int spanningTree(int V, int[][] edges) {

        // Step 1: Create adjacency list
        // adj.get(u) will contain {v, weight}
        List<List<int[]>> adj = new ArrayList<>();

        // Initialize empty list for each node
        for (int i = 0; i < V; i++) {
            adj.add(new ArrayList<>());
        }

        // Fill adjacency list from edges input
        for (int[] edge : edges) {

            int u = edge[0];       // source node
            int v = edge[1];       // destination node
            int weight = edge[2];  // edge weight

            // Since graph is undirected, add both directions
            adj.get(u).add(new int[]{v, weight});
            adj.get(v).add(new int[]{u, weight});
        }

        // Step 2: Min Heap (PriorityQueue)
        // It will always give the smallest weight edge first
        PriorityQueue<P> pq = new PriorityQueue<>(
            (a, b) -> a.weight - b.weight
        );

        // Start from node 0 with weight 0
        pq.add(new P(0, 0));

        // Track which nodes are already included in MST
        boolean[] inMST = new boolean[V];

        int totalWeight = 0; // final answer

        // Step 3: Prim's Algorithm
        while (!pq.isEmpty()) {

            // Get the edge with minimum weight
            P current = pq.poll();

            int weight = current.weight;
            int node = current.node;

            // If node already included, skip (avoid cycles)
            if (inMST[node]) {
                continue;
            }

            // Include this node in MST
            inMST[node] = true;

            // Add its edge weight to total MST cost
            totalWeight += weight;

            // Explore all neighbors of current node
            for (int[] neighborData : adj.get(node)) {

                int neighbor = neighborData[0];     // adjacent node
                int edgeWeight = neighborData[1];   // weight of edge

                // If neighbor not yet in MST, add to PQ
                if (!inMST[neighbor]) {
                    pq.add(new P(edgeWeight, neighbor));
                }
            }
        }

        // Return total weight of MST
        return totalWeight;
    }
}
