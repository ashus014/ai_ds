269. Alien Dictionary

Blind 75 #48 — Build a graph from adjacent word pairs (first differing letter gives a directed edge). Run Kahn’s algorithm for topological order. Use a boolean[][] so the same edge u → v is not added twice (duplicate edges break indegree).

Hint

Compare adjacent words → first different character gives edge → build graph → apply Kahn's topo sort

import java.util.*;

class Solution {
    public String foreignDictionary(String[] words) {

        int K = 26; // max possible chars

        // Step 1: adjacency list
        ArrayList<ArrayList<Integer>> adj = new ArrayList<>();
        for (int i = 0; i < K; i++) {
            adj.add(new ArrayList<>());
        }

        // Step 2: indegree
        int[] indegree = new int[K];

        // Step 3: mark which chars exist
        boolean[] exists = new boolean[K];

        // Same u -> v must be added once only (duplicate edges inflate indegree)
        boolean[][] hasEdge = new boolean[K][K];

        for (String word : words) {
            for (char c : word.toCharArray()) {
                exists[c - 'a'] = true;
            }
        }

        // Step 4: build edges (IMPORTANT PART)
        for (int i = 0; i < words.length - 1; i++) {

            String s1 = words[i];
            String s2 = words[i + 1];

            // invalid case: longer word before shorter prefix (impossible in sorted order)
            if (s1.length() > s2.length() && s1.startsWith(s2)) {
                return "";
            }

            int len = Math.min(s1.length(), s2.length());

            for (int j = 0; j < len; j++) {
                char c1 = s1.charAt(j);
                char c2 = s2.charAt(j);

                if (c1 != c2) {
                    int u = c1 - 'a';
                    int v = c2 - 'a';

                    if (!hasEdge[u][v]) {
                        hasEdge[u][v] = true;
                        adj.get(u).add(v);
                        indegree[v]++;
                    }
                    break; // only first diff matters
                }
            }
        }

        // Step 5: Kahn's algo (YOUR TEMPLATE — topological order)
        Queue<Integer> q = new LinkedList<>();

        for (int i = 0; i < K; i++) {
            if (exists[i] && indegree[i] == 0) {
                q.offer(i);
            }
        }

        StringBuilder result = new StringBuilder();

        while (!q.isEmpty()) {
            int u = q.poll();
            result.append((char)(u + 'a'));

            for (int v : adj.get(u)) {
                indegree[v]--;

                if (indegree[v] == 0) {
                    q.offer(v);
                }
            }
        }

        // Step 6: cycle check (all existing letters must appear in the order)
        int count = 0;
        for (boolean e : exists) if (e) count++;

        if (result.length() != count) {
            return "";
        }

        return result.toString();
    }
}
