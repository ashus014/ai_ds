Splitwise

Low-Level Design — expense splitting, groups, and balances

07 January 2026

GitHub — full source (LLD/splitwise)

Summary

Model a bill-splitting app: friends, groups, expenses with equal / unequal / percentage splits, and per-user balance sheets. Expenses can be created from a group or directly between users; controllers orchestrate CRUD and delegate split validation to a strategy-style ExpenseSplit implementation from SplitFactory.

Step 1 — Requirement gathering

  • Add friends capability
  • Add / manage group
  • Add / manage friends inside a group
  • Add / manage expense inside a group or without a group
  • Split expense: equal, unequal, percentage
  • Balance sheet per user (who owes whom, totals)

Step 2 — Object modeling

  • Splitwise — facade: wires controllers
  • User
  • Group
  • Expense
  • Split
  • BalanceSheet / UserExpenseBalanceSheet — per-user aggregates and friend-level balances

SplitFactory.java
/**
 * Returns the correct split validator for the requested type.
 */
public final class SplitFactory {

    private SplitFactory() {}

    public static ExpenseSplit getSplitObject(ExpenseSplitType splitType) {
        return switch (splitType) {
            case EQUAL -> new EqualExpenseSplit();
            case UNEQUAL -> new UnequalExpenseSplit();
            case PERCENTAGE -> new PercentageExpenseSplit();
        };
    }
}
EqualExpenseSplit.java (example)
import java.util.List;

public final class EqualExpenseSplit implements ExpenseSplit {
    @Override
    public void validateRequest(List<Split> splitList, double totalAmount) {
        if (splitList == null || splitList.isEmpty()) {
            throw new IllegalArgumentException("Splits required");
        }
        double share = totalAmount / splitList.size();
        for (Split s : splitList) {
            if (Math.abs(s.getAmount() - share) > 1e-6) {
                throw new IllegalArgumentException("Equal split: each share must be " + share);
            }
        }
    }
}

ExpenseController.java
import java.util.ArrayList;
import java.util.List;

public class ExpenseController {
    private final List<Expense> expenses = new ArrayList<>();
    private final BalanceSheetController balanceSheetController;

    public ExpenseController(BalanceSheetController balanceSheetController) {
        this.balanceSheetController = balanceSheetController;
    }

    /**
     * Creates an expense (group or non-group). Validates splits, stores expense, updates balances.
     */
    public Expense createExpense(String expenseId, String description, double expenseAmount,
                                 List<Split> splitDetails, ExpenseSplitType splitType,
                                 User paidByUser) {
        ExpenseSplit splitter = SplitFactory.getSplitObject(splitType);
        splitter.validateRequest(splitDetails, expenseAmount);

        Expense expense = new Expense(expenseId, description, expenseAmount,
                paidByUser, splitType, splitDetails);
        expenses.add(expense);
        balanceSheetController.updateUserExpenseBalanceSheet(expense);
        return expense;
    }

                    public List<Expense> getExpenses() {
        return List.copyOf(expenses);
    }
}
UnequalExpenseSplit.java
import java.util.List;

/** Split amounts must sum to {@code totalAmount}. */
public final class UnequalExpenseSplit implements ExpenseSplit {
    @Override
    public void validateRequest(List<Split> splitList, double totalAmount) {
        double sum = splitList.stream().mapToDouble(Split::getAmount).sum();
        if (Math.abs(sum - totalAmount) > 1e-6) {
            throw new IllegalArgumentException("Unequal splits must sum to total: " + totalAmount);
        }
    }
}
PercentageExpenseSplit.java
import java.util.List;

/**
 * {@link Split#getAmount()} holds percentage for each user; must sum to 100.
 * You can convert to currency in the controller after validation.
 */
public final class PercentageExpenseSplit implements ExpenseSplit {
    @Override
    public void validateRequest(List<Split> splitList, double totalAmount) {
        double pct = splitList.stream().mapToDouble(Split::getAmount).sum();
        if (Math.abs(pct - 100.0) > 1e-6) {
            throw new IllegalArgumentException("Percentages must sum to 100");
        }
    }
}

UserExpenseBalanceSheet.java
import java.util.HashMap;
import java.util.Map;

public class UserExpenseBalanceSheet {
    private final Map<User, Balance> friendBalance = new HashMap<>();
    private double totalExpense;
    private double totalPayments;
    private double totalOwe;
    private double totalGetBack;

    public Map<User, Balance> getFriendBalance() { return friendBalance; }
    public double getTotalExpense() { return totalExpense; }
    public void addTotalExpense(double v) { totalExpense += v; }
    public double getTotalPayments() { return totalPayments; }
    public void addTotalPayments(double v) { totalPayments += v; }
    public double getTotalOwe() { return totalOwe; }
    public void addTotalOwe(double v) { totalOwe += v; }
    public double getTotalGetBack() { return totalGetBack; }
    public void addTotalGetBack(double v) { totalGetBack += v; }

    private Balance balanceFor(User u) {
        return friendBalance.computeIfAbsent(u, x -> new Balance());
    }

    /** User {@code owes} paid {@code amount} toward {@code creditor}. */
    public void owe(User creditor, double amount) {
        balanceFor(creditor).addOwe(amount);
        addTotalOwe(amount);
    }

    /** User is owed {@code amount} by {@code debtor}. */
    public void getBack(User debtor, double amount) {
        balanceFor(debtor).addGetBack(amount);
        addTotalGetBack(amount);
    }
}

BalanceSheetController.java
/**
 * For each expense: payer should "get back" others' shares; each non-payer "owes" their share to payer.
 */
public class BalanceSheetController {

    public void updateUserExpenseBalanceSheet(Expense expense) {
        User paidBy = expense.getPaidBy();
        paidBy.getUserExpenseBalanceSheet().addTotalPayments(expense.getAmount());
        paidBy.getUserExpenseBalanceSheet().addTotalExpense(expense.getAmount());

        for (Split split : expense.getSplitDetails()) {
            User participant = split.getUser();
            double share = split.getAmount();
            participant.getUserExpenseBalanceSheet().addTotalExpense(share);

            if (participant.equals(paidBy)) {
                continue;
            }
            participant.getUserExpenseBalanceSheet().owe(paidBy, share);
            paidBy.getUserExpenseBalanceSheet().getBack(participant, share);
        }
    }

    public void showBalanceSheetOfUser(User user) {
        UserExpenseBalanceSheet sheet = user.getUserExpenseBalanceSheet();
        System.out.println("--- Balance sheet: " + user.getName() + " ---");
        System.out.println("Total paid (as payer): " + sheet.getTotalPayments());
        System.out.println("Total expense (your share): " + sheet.getTotalExpense());
        sheet.getFriendBalance().forEach((friend, bal) ->
                System.out.println(" vs " + friend.getName()
                        + " | owe=" + bal.getAmountOwe()
                        + " getBack=" + bal.getAmountGetBack()));
    }
}

import java.util.ArrayList;
import java.util.List;

public class Group {
    private final String id;
    private final String name;
    private final List<User> users = new ArrayList<>();
    private final List<Expense> expenses = new ArrayList<>();
    private final ExpenseController expenseController;

    public Group(String id, String name, ExpenseController expenseController) {
        this.id = id;
        this.name = name;
        this.expenseController = expenseController;
    }

    public void addMember(User u) { users.add(u); }
    public List<User> getUsers() { return List.copyOf(users); }
    public List<Expense> getExpenses() { return List.copyOf(expenses); }
    public ExpenseController getExpenseController() { return expenseController; }

    public Expense addExpense(String expenseId, String description, double amount,
                              List<Split> splits, ExpenseSplitType type, User paidBy) {
        Expense e = expenseController.createExpense(expenseId, description, amount, splits, type, paidBy);
        expenses.add(e);
        return e;
    }
}

Splitwise.java (facade)
/**
 * Top-level API: owns controllers used by the app / demo runner.
 */
public class Splitwise {
    private final UserController userController;
    private final GroupController groupController;
    private final ExpenseController expenseController;
    private final BalanceSheetController balanceSheetController;

    public Splitwise() {
        this.balanceSheetController = new BalanceSheetController();
        this.expenseController = new ExpenseController(balanceSheetController);
        this.userController = new UserController();
        this.groupController = new GroupController();
    }

    public UserController getUserController() { return userController; }
    public GroupController getGroupController() { return groupController; }
    public ExpenseController getExpenseController() { return expenseController; }
    public BalanceSheetController getBalanceSheetController() { return balanceSheetController; }
}

Note: For PERCENTAGE splits, validate percentages in PercentageExpenseSplit, then convert each line to a rupee/dollar amount before calling createExpense, or perform conversion inside ExpenseController after validation. Thread-safety, persistence, and settlement (simplifying debts) are out of scope for this LLD sketch.
