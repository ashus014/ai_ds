Tic Tac Toe

Low-Level Design — flexible board size, multi-player support, and extensible pieces

April 26, 2026

GitHub — full source (LLD/_12_ticTakToe/TicTacToe.md)

Summary

Design a scalable Tic-Tac-Toe game that supports a variable board size (N x N), multiple players, and extensible playing pieces. The system handles game state management, turn-based logic, and win-condition detection.

Requirements & Goals

  1. Flexible Board Size: The game should support board sizes like 3x3, 4x4, up to NxN.
  2. Multiple Players: Support for 2 or more players using different pieces (X, O, etc.).
  3. Piece Extensibility: Playing pieces should be extensible to support different shapes, colors, or rendering logic in the future.
  4. Win Detection: Efficiently check for win conditions in rows, columns, and diagonals.
  5. Draw Detection: Handle cases where the board is full and no player has won.
  6. Turn Management: Maintain a queue of players to manage turns sequentially.

        Player-->>Game: return Move
        Game->>Board: addPiece(row, col, piece)

        Game->>Board: checkGameStatus()
        Board-->>Game: return GameStatus
    end
    
    Game->>Board: printBoard()
    Game->>Main: displayResult()

Major Objects

  1. PlayingPiece: Enum or Class representing the piece type (X, O).
  2. Board: Manages the N x N grid and piece placement.
  3. Player: Abstract base class for different player types (Human, AI).
  4. Move: Captures the coordinates and piece for a single turn.
  5. GameStatus: Tracks the current state (In Progress, Win, Draw).
  6. TicTacToeGame: Orchestrates the game flow and player turns.

Board.java

package LLD.ticTakToe;

import java.util.ArrayList;
import java.util.List;

public class Board {
    private int size;
    private PlayingPiece[][] board;

    public Board(int size) {
        this.size = size;
        this.board = new PlayingPiece[size][size];
    }

    public boolean addPiece(int row, int col, PlayingPiece piece) {
        if (row < 0 || row >= size || col < 0 || col >= size || board[row][col] != null) {
            return false;
        }
        board[row][col] = piece;
        return true;
    }

    public List getFreeCells() {
        List freeCells = new ArrayList<>();
        for (int i = 0; i < size; i++) {
            for (int j = 0; j < size; j++) {
                if (board[i][j] == null) {
                    freeCells.add(new int[]{i, j});
                }
            }
        }
        return freeCells;
    }

    public GameStatus checkGameStatus() {
        // Check rows, columns and diagonals
        for (int i = 0; i < size; i++) {
            if (checkLine(i, 0, 0, 1)) return getWinnerStatus(board[i][0]);
            if (checkLine(0, i, 1, 0)) return getWinnerStatus(board[0][i]);
        }
        if (checkLine(0, 0, 1, 1)) return getWinnerStatus(board[0][0]);
        if (checkLine(0, size - 1, 1, -1)) return getWinnerStatus(board[0][size - 1]);

        if (getFreeCells().isEmpty()) return GameStatus.DRAW;
        return GameStatus.IN_PROGRESS;
    }

    private boolean checkLine(int startRow, int startCol, int rowDelta, int colDelta) {
        PlayingPiece first = board[startRow][startCol];
        if (first == null) return false;
        for (int i = 1; i < size; i++) {
            if (board[startRow + i * rowDelta][startCol + i * colDelta] != first) return false;
        }
        return true;
    }

    private GameStatus getWinnerStatus(PlayingPiece piece) {
        return piece == PlayingPiece.X ? GameStatus.PLAYER_X_WINS : GameStatus.PLAYER_O_WINS;
    }
}
TicTacToeGame.java

package LLD.ticTakToe;

import java.util.Deque;
import java.util.LinkedList;
import java.util.List;
import java.util.Scanner;

public class TicTacToeGame {
    private Deque players;
    private Board gameBoard;
    private GameStatus gameStatus;
    private Scanner scanner;

    public TicTacToeGame(Scanner scanner) {
        this.players = new LinkedList<>();
        this.gameStatus = GameStatus.IN_PROGRESS;
        this.scanner = scanner;
    }

    public void startGame() {
        while (gameStatus == GameStatus.IN_PROGRESS) {
            gameBoard.printBoard();
            Player currentPlayer = players.pollFirst();
            
            boolean moveSuccess = false;
            while (!moveSuccess) {
                Move move = currentPlayer.makeMove(gameBoard, scanner);
                moveSuccess = gameBoard.addPiece(move.getRow(), move.getCol(), move.getPiece());
            }

            players.addLast(currentPlayer);
            gameStatus = gameBoard.checkGameStatus();
        }
        displayResult();
    }
}
